SQL Code Example


SELECT

    mrn = pax.alias

    , mrn_type = uar_get_code_display (pax.alias_pool_cd )

    ;, pax.alias_pool_cd

, PAX_CONTRIBUTOR_SYSTEM_DISP = UAR_GET_CODE_DISPLAY(PAX.CONTRIBUTOR_SYSTEM_CD)

    , personnel_name = pr.name_full_formatted

    , action_type = uar_get_code_display (pa.action_type_cd )

    , pa.action_dt_tm "@MEDIUMDATETIME"

    , personnel_id = pr.person_id

    , pi.pregnancy_id

    , pi.preg_start_dt_tm "@MEDIUMDATETIME"

    , pi.preg_end_dt_tm "@MEDIUMDATETIME"

    , pi.active_ind

    , pi.beg_effective_dt_tm "@MEDIUMDATETIME"

    , pi.end_effective_dt_tm "@MEDIUMDATETIME"

FROM

    pregnancy_instance   pi

    , prsnl   pr

    , pregnancy_action   pa

    , person_alias   pax

PLAN pax

   WHERE pax.person_alias_type_cd = 10

   AND pax.active_ind = 1

   AND pax.end_effective_dt_tm > cnvtdatetime (curdate ,curtime3)

   AND pax.alias_pool_cd in  (18991198.00)

   and PAX.CONTRIBUTOR_SYSTEM_CD in( 15058181, 19418947 ); CUMH_ADT, COGA_ADT

   JOIN pi

   WHERE pi.person_id = pax.person_id

   AND pi.active_ind = 1

   AND pi.historical_ind = 0

   AND pi.preg_end_dt_tm < cnvtdatetime (curdate ,curtime3)

   JOIN pa

   WHERE pa.pregnancy_id = pi.pregnancy_id

   AND pa.action_type_cd =    15052788.00

   and pa.action_dt_tm between CNVTDATETIME(CURDATE-7, CURTIME3) AND CNVTDATETIME(CURDATE, CURTIME3)

   JOIN pr

   WHERE pi.updt_id = pr.person_id  

   and pr.person_id = 1.00

ORDER BY

    pa.action_dt_tm   DESC

WITH MAXREC = 5000, noncounter ,format ,separator = "", TIME=30
